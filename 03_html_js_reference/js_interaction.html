Other interesting interactions

Button as toggle 

<p>
<button id="buttonAsToggle">Show</button>
<span id="buttonAsToggleResult"></span>
<script type="text/javascript">
// INTERACTION
document.querySelector('#buttonAsToggle').addEventListener('click', function() {
    showContent()
  }
);

// TOGGLE FUNCTION
var isShown = false;

function showContent() {
  if (isShown) {
    document.querySelector("#buttonAsToggleResult").innerHTML = "";
    document.querySelector("#buttonAsToggle").innerHTML = "Show";
  } else {
    document.querySelector("#buttonAsToggleResult").innerHTML = "HERE I AM üëª";
    document.querySelector("#buttonAsToggle").innerHTML = "Hide";
  }
  isShown = !isShown;
}
</script>
